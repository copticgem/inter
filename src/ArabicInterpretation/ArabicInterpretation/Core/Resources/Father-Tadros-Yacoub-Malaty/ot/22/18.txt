يدعونا سفر الحكمة إلى العزلة المقدسة التي فيها ينسحب الإنسان تحت قيادة روح الله القدوس إلى أعماقه، ليعيد تقييم نفسه، وتقييم عمل الله فيه، وإعادة النظر إلى كل شيء بعينيّ رب المجد يسوع، الحكمة الإلهي. {{p}}وفي نفس الوقت يحذرنا من العزلة الشريرة التي فيها ينسحب الإنسان من الجماعة المقدسة، حاسبًا نفسه أفضل من الجميع، معتمدًا على فكره الذاتي، رافضًا المشورة المقدسة والحوار المقدس.{{p}}{{g}}{{gr}}{{gc}} ١. العزلة الشريرة{{/gc}}{{gc}} ١-٣.{{/gc}}{{/gr}}{{gr}}{{gc}} ٢. اللسان نهر متدفق{{/gc}}{{gc}} ٤-٩.{{/gc}}{{/gr}}{{gr}}{{gc}} ٣. الثروة الروحية{{/gc}}{{gc}} ١٠-١١. {{/gc}}{{/gr}}{{gr}}{{gc}} ٤. بركة التواضع{{/gc}}{{gc}} ١٢.{{/gc}}{{/gr}}{{gr}}{{gc}} ٥. عطية الاستماع{{/gc}}{{gc}} ١٣.{{/gc}}{{/gr}}{{gr}}{{gc}} ٦. عطية الرجاء{{/gc}}{{gc}} ١٤.{{/gc}}{{/gr}}{{gr}}{{gc}} ٧. مزيد من الحكمة{{/gc}}{{gc}} ١٥.{{/gc}}{{/gr}}{{gr}}{{gc}} ٨. علاج الخصومات{{/gc}}{{gc}} ١٦-١٩.{{/gc}}{{/gr}}{{gr}}{{gc}} ٩. ثمار اللسان{{/gc}}{{gc}} ٢٠-٢١.{{/gc}}{{/gr}}{{gr}}{{gc}} ١٠. اختيار شريكة الحياة{{/gc}}{{gc}} ٢٢.{{/gc}}{{/gr}}{{gr}}{{gc}} ١١. الفقير والغني{{/gc}}{{gc}} ٢٣.{{/gc}}{{/gr}}{{gr}}{{gc}} ١٢. الصداقة{{/gc}}{{gc}} ٢٤.{{/gc}}{{/gr}}{{gr}}{{gc}}من وحي الأمثال ١٨{{/gc}}{{/gr}}{{/g}}{{p}}{{d}}{{p}}{{v1}}{{v2}}{{v3}}{{t}} ١. العزلة الشريرة{{/t}}{{p}}{{b}} "المعتزل يطلب شهوته بكل مشورة يغتاظ" {{/b}} [ع ١].{{p}}يحدثنا الرسول يهوذا عن هؤلاء المعتزلين الذين في تشامخ يتعالون عن إخوتهم، ويطلبون تحقيق شهواتهم وملذاتهم. "هؤلاء هم المعتزلون بأنفسهم لا روح لهم (ليس لهم الروح القدس)" (يه ١٩). هؤلاء يعتزلون الحق الإلهي والجماعة المقدسة لمجد أنفسهم أو لإشباع شهواتهم، وليس لمجد الله، هؤلاء عنيدون متعصبون لأفكارهم الذاتية.{{p}}ما يشغل فكر هؤلاء المعتزلين هو إبراز مهارتهم أو ذكائهم أو عملهم، وليس بنيان نفوسهم وتقدمهم حتى في دراستهم لكلمة الله.{{p}}هؤلاء المعتزلون لحساب ملذاتهم، يرفضون الحكمة الحقيقية، خلاف المعتزلين الشر ومشاركة الأشرار شرورهم لأجل نقاوتهم في الرب خلال عضويتهم الكنسية. وكما يقول الرسول بولس: "لذلك أخرجوا من وسطهم، واعتزلوا يقول الرب، ولا تمسوا نجسًا فأقبلكم، وأكون لكم أبًا وأنتم تكونون لي بنين وبنات يقول الرب القادر على كل شيء" (٢ كو ٦: ١٧-١٨).{{p}}* هؤلاء (المعتزلون) هم الذين يفصلون المؤمنين الواحد عن الآخر، وذلك بتأثير عدم إيمانهم. هؤلاء لا يستطيعون أن يميزوا المقدسات من جانب والكلاب من جانب آخر[٥٧٣].{{p}}{{b}} القديس إكليمنضس السكندري{{/b}}{{p}}* عدو الوحدة ليس له شركة في حب الله. أولئك الذين هم خارج الكنيسة، ليس لهم الروح القدس[٥٧٤].{{p}}القديس أغسطينوس{{p}}يليق بالمؤمن أن يعتزل الشر نفسه، ليس بألا يرتكبه فحسب، بل ولا يسمح لفكره بالحوار معه.{{p}}* سأل إخوةٌ شيخًا بخصوص القولين السابقين قائلين: "هل إذا لم يكمِّل الإنسان الأفكار النجسة تبطُل"؟ {{b}}فقال الشيخ:{{/b}} ”إنه يقصد ليس أن لا يتممها بجسده فحسب، بل أيضًا أن لا يتناقش معها، لأن أنبا بيمين يعلم أن الأفكار لا تكمل بالفعل إلاّ بالتفاوض معها وطاعتها، فإذا ترك الإنسان السبب والعلّة المؤدّية إلى الفعل فهو ينعتق من الأفكار ويغلبها، وهكذا بعد زمنٍ من الجهاد يتنقّى قلبه ويستنير ذهنه ويخلص من الأوجاع ويفرح بالله".{{p}}* سأل أبّا أمون الذي من "رايثو" أنبا بيمين عن الأفكار النجسة والشهوات الباطلة التي تهاجم الإنسان، فقال له أنبا بيمين: "من اختصاص الشيطان أن يزرعها فينا، أما عملنا نحن فهو أن لا نقبلها".{{p}}{{b}} فردوس الآباء{{/b}}{{p}}* يا أولادي، فِرُّوا من الخطية واصبروا إلى الموت في حفظ وصايا الرب، ولا تقبلوا مشورة العدو من جهة كسر أي وصية مهما كانت صغيرة، لأنّ كسر أي وصية صغيرة كانت أم كبيرة يُغضِب الله. إنني أريد أن تكون نفوسكم، يا أولادي، مسكنًا دائمًا لله حتى تتفكّروا على قريبكم بالخير دائمًا ولا يكون فيكم مَنْ يذكر الشر لأخيه أو يتحرك بالبغضة عليه، فإنّ القلب الذي يتفكّر بالشر والبغضة لا يمكن أن يكون مسكنًا لله.{{p}}{{b}} القديس مقاريوس الكبير{{/b}}{{p}}{{b}} "الجاهل لا يسر بالفهم،{{/b}}{{p}}{{b}} بل بكشف قلبه" {{/b}}[ع ٢]{{p}}إن كان المعتزل لأجل إشباع شهواته لا يقبل مشورة حكيمة، بل يعتد برأيه وحده، فإنه لا يجد مسرة في الفهم، إنما مسرته أن يحقق ما في قلبه. مثل هذا الجاهل لا يقبل الحوار، ولا يهتم بما ينفعه وما يبني الآخرين.{{p}}* في كل ما تفعله خُذ لنفسك مشورةً لأنه مكتوبٌ: "من الحماقة العمل بغير مشورة" (قارن أم١٥: ٢٢)، وإذا سألك أحدٌ أجبه وإلاّ فالصمت أفضل.{{p}}{{b}} أنبا بيمين{{/b}}{{p}}* زار بعض الإخوة ومعهم علمانيون الأب "فيليكس" وتوسلوا إليه أن يقول لهم كلمة، ولكن الشيخ ظلّ صامتًا. ولما ألحّوا عليه لمدة طويلة قال لهم: "أتُريدون أن تسمعوا كلمة؟"{{p}}فقالوا: "نعم أيها الأب". {{p}}فقال لهم: "لا يوجد كلام بعد في هذه الأيام، لأنه عندما كان الإخوة يسألون مشورة الشيوخ ويعملون بما يُقال لهم، كان الله يُلهِم الآباء بما يقولون، أما الآن فلأنهم يسألون ولا يفعلون بما يسمعونه، فقد سحب الله نعمة الكلام من الشيوخ، ولا يجدون شيئًا يقولونه، لأنّ ليس مَنْ يعمل، لأنّ المرتل يقول: "الرب من السماء أشرف على بني البشر لينظر هل مِنْ فاهمٍ طالب الله!" (مز١٤: ٢). {{p}}فلما سمع الإخوة ذلك تنهّدوا وقالوا: "صلِّ من أجلنا يا أبانا".{{p}}{{b}} فردوس الآباء{{/b}}{{p}}{{b}} "إذا جاء الشرير جاء الاحتقار{{/b}}{{p}}{{b}} وأيضًا ومع الهوان عار" {{/b}}[ع ٣]{{p}}إذ يتمسك الشرير الجاهل برأيه، ويرفض الحوار، يجلب لنفسه خزيًا وتعييرًا. فإنه إذ يستخف بآراء الغير لا يجد الآخرون فيه مسرة. أما من يحترم الآخرين ويُقدر أفكارهم، ويحاورهم بالحب وفي تواضع، فإن روح الله القدوس العامل في المتواضعين يعطيه كرامة في أعين إخوته والمحيطين به.{{p}}إن كانت الخطية هي عار الشعوب (أم ١٤: ٣٤)، فهي خزي كل إنسان يُصر على ارتكابها. الخطية هي الطريق التي تحدر الشعوب كما الأفراد إلى عار الهاوية.{{p}}إذ امتدت يد فشحور بن أمير الكاهن ليضرب إرميا النبي، ويجعله في المقطرة، صدر الحكم الإلهي ضده بأن يمتلئ خوفًا ورعبًا لما يحل به وبمحبيه، ويسقطون في السبي البابلي، وهناك يدفنون في عارٍ وخزيٍ (إر ٢٠: ١-٦).{{p}}* صدقوا هذا بتقوى وثباتٍ: الله لا يتخلى عن شخص ما لم يترك الشخص الله فعلًا. وبالرغم من ارتكاب الشخص خطايا خطيرة مرة ومرتين وثلاثًا، فإن الله يبقى يطلبه، كما يقول النبي أن برجوعه يحيا (خر ٣٣: ١١). {{p}}على أي الأحوال، إذ يستمر في خطاياه، يقدم فيه اليأس بسبب كثرة خطاياه، وتحدث قسوة بسبب اليأس. بينما ييأس الناس الملهمون بسبب خطاياهم إذ هي صغيرة، فإن هذه المعاصي البسيطة تتزايد، بل ويُضاف إليها جرائم، وتصير كومة تبتلعهم، وإذ يحدث هذا يتحقق المكتوب: "مع الشر يأتي العار"[٥٧٥].{{p}}{{b}}الأب قيصريوس أسقف آرل{{/b}}{{p}}* مكتوب: "إذا جاء الخاطئ إلى أعماق الشر يسلك باستخفاف" (أم ١٨: ٣). إنهم (الأشرار) لا يؤمنون كما ترون، فلا يمكن أن يُغفر لهم ما قد فعلوه، بهذا الحجر من اليأس يغطسون في أعماق أكثر مما كانوا عليه[٥٧٦]. {{p}}القديس أغسطينوس{{p}}* عندما يسقط الأشرار إلى أعماق الشر، يفقد كل إحساسٍ بالوقار". إنه لأمر مرعب كما ترون أيها الأعزاء المحبوبون، أمر مرعب هو السقوط بين أنياب الشيطان، أعني أن النفس تصير كمن سقطت في شبكة، ومثل خنزير برِّي يسقط في شرك من الوحل، اقتنصته اللذة، يرتد إلى عاداته الشريرة، يفقد ملء الإحساس برائحة الخطية الكريهة. لهذا يلزمنا أن نستيقظ ونأخذ حذرنا، فلا نسمح لشيطان شرير أن يجد من البداية مدخلًا، فيضع غمامة على عقلنا، ويعمي بصيرة ذهننا الحادة، وكأنه يسلبنا نور الشمس، فلا نري أشعة شمس البرّ، فنسقط في الهاوية[٥٧٧].{{p}}{{b}}القديس يوحنا الذهبي الفم{{/b}}{{p}}* العقل الشرير دائمًا في ألمٍ وتعب، إذ أنه إما يعاني من مصائب تحل عليه أو يخشى من حلولها عليه من آخرين. وبينما يخطط ضد الأقرباء يكون هو بالأكثر في رعبٍ من أن يخطط أقرباؤه ضده... حتى عندما يكون في سلام يرتاب من الخطط التي يتعامل معها بمكرٍ، حاسبًا أنه لا يوجد إنسان ما يسلك معه بأمانة[٥٧٨].{{p}}{{b}}البابا غريغوريوس (الكبير){{/b}}{{p}}* كلما تزايدت تلك الخطايا الخطيرة، تعتاد، النفس عليها بالأكثر، وتستهين بها. إذ يُقال: "عندما بلغ الشرير إلى عمق الشرور، يفكر فيها باستخفاف"[٥٧٩].{{p}}{{b}}الأب يوحنا الدمشقي{{/b}}{{p}}{{d}}{{p}}{{v4}}{{v5}}{{v6}}{{v7}}{{v8}}{{v9}}{{t}} ٢. اللسان نهر متدفق{{/t}}{{p}}{{b}} "كلمات فم الإنسان مياه عميقة،{{/b}}{{p}}{{b}} نبع الحكمة نهر متدفق" {{/b}}[ع ٤]{{p}}إذ يسكن روح الله القدوس في المؤمن الحقيقي بربنا يسوع المسيح، تفيض من بطنه أنهار مياه حيَّة (يو ٧: ٣٨)، هذا هو نبع الحكمة الذي لن ينضب قط. يعاتب الله شعبه قديمًا قائلًا: "لأن شعبي عمل شرين، تركوني أنا ينبوع المياه الحية، لينقروا لأنفسهم آبارًا آبارًا مشققة لا تضبط ماءً" (إر ٢: ١٣).{{p}}* الكلمة في قلب الإنسان {{b}}مياه عميقة{{/b}}، ونهر ونبع متدفق. يقصد سليمان بالمياه العميقة "مياه وفيرة". يمكن دومًا أن تفيض أنهارًا عديدة عوض نهر واحد. أو يقول: إنها تحوي ما هو خفي وعميق فيها. كما أن المياه لا تُقاس، هكذا الكلمة في قلب الشخص الذي يحيا في الرب فهي بلا حدود. لهذا يستخدم "الكلمة" عند المعرفة. هذا هو الشخص الذي في قلبه نبع ماء قد تدفق عند كلمات يسوع. يتحدث عنها (سليمان) أنها متدفقة، تمطر على الحقول، وترويها، فتصير مخصبة[٥٨٠].{{p}}{{b}}القديس يوحنا الذهبي الفم{{/b}}{{p}}{{b}} "رفع وجه الشرير ليس حسنًا لأخطاء الصديق في القضاء" {{/b}} [ع ٥].{{p}}إذ يود الله أن تكون الكنيسة أيقونة السماء، والبشر سفراء عنه، يشدد على عدم المحاباة في القضاء، فلا يجوز تبرير الشرير، وتلفيق تُهم للصديقين الأبرار، كلاهما مكرهة للرب الديان العادل. جاء في سفر التثنية: "وأمرت قضاتكم في ذلك الوقت، قائلًا: "اسمعوا إخوتكم، واقضوا بالحق بين الإنسان وأخيه ونزيله. لا تنظروا إلى الوجوه في القضاء، للصغير كالكبير تسمعون. لا تهابوا وجه إنسان، لأن القضاء لله" (تث ١: ١٦-١٧).{{p}}* يلزم الذين يسمعون القضايا أن يحكموا بالعدل، ولا يقبلوا رشاوى على حساب البريء، "لأن الرشوة تعمي أعين الحكماء، وتعوِّج كلام الصديقين" (تث ١٦: ١٩). لئلا وهم يطلبون المال يفقدون النفس. لا يحصل أحد مكسب ظالم دون أن تحل به خسارة عادلة. حيث يوجد المكسب تكون الخسارة، المكسب في خزانة المال، ولكن الخسارة في الضمير[٥٨١].{{p}}{{b}} الأب قيصريوس أسقف آرل{{/b}}{{p}}هذا ما ارتكبه الشعب اليهودي حين رفعوا وجه الشرير باراباس لا لشيء إلا ليحكموا على رب المجد يسوع البار بالصلب ظلمًا.{{p}}{{b}} "شفتا الجاهل تداخلان في الخصومة،{{/b}}{{p}}{{b}} وفمه يدعو بضربات" {{/b}}[ع ٦].{{p}}لا عمل لشفتي الجاهل (الأحمق) سوى الدعوة لإثارة خصومات، فإن كان فم البار ينبوع مياه حية، ممسوحة بمسحة الروح القدس، تطفئ نيران الغضب، وتسكب سلامًا وبرودة على النفوس الثائرة، فعلى العكس فم الأحمق مصدر متاعب وقلاقل، يجد لذة في إثارة الخصومات والمنازعات.{{p}}{{b}} "فم الجاهل مهلكة له،{{/b}}{{p}}{{b}} وشفتاه شرك لنفسه" {{/b}}[ع ٧]{{p}}هذه الخصومات التي يثيرها فم الأحمق في حياة المحيطين به ترتد إليه، فيعاني من الهلاك، ويسقط في الشباك التي نصبها لإخوته.{{p}}{{b}} "كلام النمَّام مثل لقم حلوة،{{/b}}{{p}}{{b}} وهو ينزل إلى مخادع البطن" {{/b}}[ع ٨]{{p}}يقدم النمَّام كلماته كلقمة حلوة، فيجد البعض لذة في الاستماع إليه، ربما كنوعٍ من التسلية أو لمجاملة النمَّام، لكن المستمع يعاني من هذه الكلمات، إذ تنزل كما في مخادع بطنه، لا يعرف الخلاص من سمومها.{{p}}* {{b}} قال أنبا {{/b}}{{b}} مرقس:{{/b}} "كل ما تقوله عن أخيك من ورائه ولا تستطيع أن تقوله أمامه هو نميمة ومذمّة، وكل اهتمامٍ لا يؤدّي إلى صلاح العبادة هو اهتمام دنيوي".{{p}}* {{b}} سُئِل (الأب {{/b}}{{b}} إشعياء{{/b}}{{b}}){{/b}} عما هي النميمة، فأجاب: "هي الجهل بمجد الله وبغضة الآخرين".{{p}}{{b}} فردوس الآباء{{/b}}{{p}}{{b}} "أيضًا المتراخي في عمله هو أخو المسرف" {{/b}} [ع ٩].{{p}}كما أن الإنسان المسرف أو المبذِّر يبدد أمواله وينفقها في غير مكانها، هكذا المتهاون أو المتكاسل أو المتراخي يبدد وقته الثمين، يحسبه كأنه تراب لا قيمة له.{{p}}* كذلك إن لم تحاول الأيدي القوية أن تحمل الأعمال الصالحة التي بدأتها لتبلغ إلى حيث الكمال، فإن مجرد التراخي في الاجتهاد لا يتوافق مع ما تم إنجازه في البداية. لهذا يقول سليمان: {{b}}"أيضًا المتراخي في عمله هو أخو المسرف"{{/b}} (أم ١٨ : القوة[٥٨٤]. {{p}}القديس كيرلس الأورشليمي{{p}}* "اَخبرني يا من تحبه نفسي". إنني أدعوك هكذا، لأن اسمك فوق كل اسم (في ٢: ٩). إنه لا يوصف، وغير مدرك بالعقل البشري. لذلك فإن اسمك يكشف عن صلاحك، علاقتي بك روحية[٥٨٥].{{p}}{{b}} القديس غريغوريوس النيسي{{/b}}{{p}}{{b}} "ثروة الغني مدينته الحصينة،{{/b}}{{p}}{{b}} ومثل سور عال في تصوره" {{/b}}[ع ١١].{{p}}بينما يتحصن المؤمن الحقيقي باسم الرب كحصنٍ منيع، العالم عاريًا. وكما يقول السيد المسيح: "ويل لكم أيها الأغنياء، لأنكم قد نلتم عزاءكم، ويل أيها الشباعى لأنكم ستجوعون" (لو ٦: ٢٤). كما قال: "ما أعسر المتَّكلين على الأموال إلى ملكوت الله، مرور جَمل من ٣)، وكانت آذان كل الشعب نحو سفر الشريعة، وبارك عزرا الرب الإله العظيم، وأجاب جميع الشعب آمين، آمين. قال نحميا وعزرا الكاهن واللاويون المفهمون الشعب: "لا تحزنوا، {{b}}لأن فرح الرب هو قوتكم{{/b}}" (نح للفقراء والمحتاجين والمرذولين، كما يقدم العطاء للأسرة، ويقدم بالحب للرؤساء كما المرؤوسين، ويقدم لمن يحملون خصومة أو سوء فهم من جهة الإنسان. لكن الخط الرئيسي في العطية ليست القيمة المادية، ولا بهدف الرشوة، إنما بالتعبير عن الحب والخضوع والعرفان بالجميل.{{p}}العالم في حاجة أن يأخذ، والكنيسة في حاجة أن تعطي. تعطي ثمار الروح من محبة وفرح وسلام... فابتسامة الإنسان النابعة من أعماق القلب بعمل روح يلتزم أن يفحص الأمر مع الطرف الآخر، حتى وإن كان الإنسان واثقًا في صدق الأول. لأن كل طرف إنما يتطلع إلى الأمر من وجهة نظره، مبررًا تصرفاته، ملقيًا باللوم على الطرف الآخر.{{p}}ليس من الحكمة ولا من العدالة أن نصدر حكمًا ولو في فكرنا دون فحص الأمر من كل جوانبه.{{p}}* من يتهم نفسه في بداية حديثه محق (راجع أم ١٨: ١٧). ها أنتم ترون أن الاعتراف له قوة فاعلية عظمى لتصحيح الأخطاء. هكذا فإن إنكار بالحب.{{p}}{{b}} "الموت والحياة في يد اللسان،{{/b}}{{p}}{{b}} وأحباؤه يأكلون ثمره" {{/b}}[ع ٢١]{{p}}باللسان نزرع إما حبًا يهب حياة، أو خصومات ومنازعات تقدم موتًا. ليس من كلمة ننطق بها لا تأتي بثمرٍ، حتى وإن كانت عابرة.{{p}}* احذر اللسان الثرثار والأذنين المتلهفتين لسماع الأخبار. لا تحط من شأن الآخرين، ولا تصغِ إلى من يحط من شأن الغير.{{p}}* السيِّدة العجوز المهذارة، والشيخ الذي يتحدث بخرافات، والسوفسطائي في كلماته، هؤلاء جميعًا يصادرون الكتب المقدسة، ويمزقونها إلى قطع، قد يُعلِّمون بها دون أن يتعلَّموها.{{p}}* كلما أخطأ اللسان، صار بالأكثر بائسًا![٥٩٦]{{p}}{{b}} القديس جيروم{{/b}}{{p}}* كيف يمكن للجسد أن يكون ذبيحة؟ لا تدع عينك تنظر إلى شيءٍ شريرٍ، فتصير بالفعل ذبيحة (مت ٥: ٢٩، ٦: ٢٢، ١٨: ٩) . ليت لسانك لا ينطق بكلمة دنسة فيصير تقدمة. ولا تفعل يدك أمرًا رديئًا، فتصير محرقة كاملة. ولكن حتى هذا لا يكفي إذ نلتزم بممارسة أعمال صالحة أيضًا (٢ تي ٢: ٢١). تلتزم اليد بتقديم صدقات، {{b}}والفم أن يبارك الذين يلعنوه{{/b}}، والأذنان أن تجدا وقتًا لتصغي لقراءة الكتاب المقدس. هذه هي بكور كل الأعمال[٥٩٧]. {{p}}{{b}} القديس يوحنا الذهبي الفم{{/b}}{{p}}تقديس اللسان لا يتحقق بتقديس الكلمات الصادرة عنه فحسب، وإنما بتقديس الصمت، حين يترك الأعمال المقدسة تتكلم بلغة الحب والحنو، فتبلغ كلماتها إلى قلوب السامعين.{{p}}* قال الأب بيمين عن الأب نستروس إنه كان مثل الحيّة النحاسية التي رفعها موسى النبي لأجل شفاء الشعب، فقد كان يمتلك كل فضيلة، ودون أن يتكلم كان يشفي كل واحد.{{p}}{{b}} فردوس الآباء{{/b}}{{p}}{{d}}{{p}}{{v22}}{{t}} ١٠. اختيار شريكة الحياة{{/t}}{{p}}{{b}} "من يجد زوجة يجد خيرًا،{{/b}}{{p}}{{b}} وينال رضى من الرب" {{/b}}[ع ٢٢]{{p}}ليست كل زوجة (أو زوج) هو من الله، لكن من يسلم حياته في يد الرب، ويطلب ما هو لبنيانه وبنيان الجماعة، يرشده الرب إلى الزوجة (أو الزوج) المقدسة، وتكون الأسرة كنيسة مرضية ومحبوبة للرب، موضع سروره.{{p}}* ليس هناك شيء أثمن، ولا يمكن أن يوجد شيء أثمن من أن يُحَّب الإنسان من زوجته ويحبِّها! {{p}}يقول الكتاب: "زوجة تتّفق مع رجلها" (كإحدى البركات الثلاث التي ذكرها الحكيم) (ابن سيراخ ٢٥: ١-٢)، لأنه حيث يوجد هذا (الحب) يوجد الغنى ويفيض الرخاء، وحيث ينعدم هذا لا يفيد أي شيء آخر، بل تسير كل الأمور في طريق خاطئ، وتصبح الحياة بؤسًا وقلقًا. {{p}}ليتنا نطلب هذا قبل كل شيء! {{p}}من يطلب المال لا يطلب شيئًا. لتطلب ما يدوم! {{p}}لا تطلب الزوجة من بين الأغنياء، لئلا زيادة الغنى من جانبها يسبب عجرفة. {{p}}{{b}} يليق بنا ألا نطلب الزوجة من أجل ثروتها، بل نطلبها شريكة{{/b}} لنا في الحياة من أجل التدبير لتكون معينة من أجل إنجاب الأبناء. {{p}}لقد أعطى الله المرأة لا لتجلب مالًا بل لتكون معينة![٥٩٨] {{p}}* الزوج والزوجة هما واحد كما أن الخمر والماء هما واحد عند امتزاجهما معًا. كما أن الشريك غير المؤمن يفسد المؤمن. لهذا السبب فإن الذي لم يتزوج بعد يلزمه بكل حرص إما أنه لا يتزوج نهائيًا أو يتزوج في الرب[٥٩٩].{{p}}العلامة أوريجينوس{{p}}{{d}}{{p}}{{v23}}{{t}} ١١. الفقير والغني{{/t}}{{p}}{{b}} "بتضرعات يتكلم الفقير،{{/b}}{{p}}{{b}} والغني يجاوب بخشونة{{/b}}" [ع ٢٣]{{p}}كثيرًا ما يعتمد الغني على مقتنياته، ويتكلم بروح التشامخ كمن يتكئ على ثروته، لكن يليق به أن يدرك أن الرب نفسه ينصت إلى تضرعات الفقير المظلوم وصرخاته الخفية.{{p}}{{d}}{{p}}{{v24}}{{t}} ١٢. الصداقة{{/t}}{{p}}{{b}} "المكثر الأصحاب يخرِّب نفسه،{{/b}}{{p}}{{b}} ولكن يوجد محب ألزَق من الأخ" [٢٤]{{/b}}{{p}}يخشى الحكيم من كثرة الأصدقاء غير الحكماء، فيفسدون وقته، لكن مع تقديم الحب للكل، خاصة الزملاء، يلزم أن يختار الإنسان صديقًا روحيًا يكون ألزق من أخيه حسب الجسد.{{p}}* تصور دائرة تخرج من مركزها أشعة أو خطوط. فبقدر ما تبتعد الخطوط عن المركز تفترق عن بعضها البعض... وبالعكس كلما اقتربت من المركز تقاربت نحو بعضها البعض. افترض أن هذه الدائرة في العالم ومركز الدائرة هو الله. والخطوط من المركز إلى المحيط أو من المحيط إلى المركز هي طريق حياة البشر، فإننا نجد نفس الأمر، فبقدر ما يتحرك القديسون في داخل الدائرة تجاه المركز راغبين في الاقتراب من الله، يقترب كل منهم نحو الآخر[٦٠٠].{{p}}{{b}} القديس دوروثيؤس{{/b}}{{p}}* لنصادق الصديقين ونعيش معهم فترات طويلة، لا ليحسبنا الآخرين أننا قديسون مثلهم، بل بقصد الإقتداء بهم ونوال بركتهم. فصداقة المجاهدين تلهب القلب بالغيرة والجهاد. لكن لندرك أنهم بشر، معرضون للسقوط، وليسوا آلهة معصومين من الخطأ، بهذا فإن أخطأوا لا نخطئ مثلهم ولا نيأس نحن من خلاصنا!{{p}}* كما أن الذين يجالسون باعة المسك والأطياب العبقة يكتسبون الروائح الذكية، هكذا ينبغي علينا أن نلازم الحكماء والمعلمين وأرباب الفضيلة، لنقتدي بمثالهم في الصالحات.{{p}}{{b}} القديس يوحنا الذهبي الفم{{/b}}{{p}}{{d}}{{p}}{{t}} من وحي أمثال ١٨{{/t}}{{t}} لأحيا في بيتك واعتزل الشر!{{/t}}{{p}}* أي ملجأ لي سواك، {{p}}ففي بيتك أقيم ومع أبنائك أتمتع بك.{{p}}أحيا بروح الحب والوحدة، {{p}}فأنعم بخلاصك.{{p}}واعتزل الشر والباطل.{{p}}يتسع قلبي لكل البشر، {{p}}لكن لا شركة لي مع الفساد المحطم!{{p}}* في بيتك أقبل المشورة الحكيمة المقدسة.{{p}}أنعم بحوار الحب مع إخوتي.{{p}}وأتلمس عمل روحك القدوس في كنيستك.{{p}}* روحك القدوس واهب الوحدة والحكمة يقود كنيستك.{{p}}يحول ألسنتنا إلى أنهار مياه عميقة،{{p}}وينبوع حكمة صادقة.{{p}}* كنيستك أيقونة السماء!{{p}}وشعبك سفراء مملكتك.{{p}}هب للبٌر والعدل أن يصيرا ناموس كل عضوٍ فيها.{{p}}وهب لكل قلب ألا يحمل بغضة وكراهية،{{p}}فلا يكون للخصومة موضع فيها.{{p}}ولا لكلمة جارحة تصدر عن شفتي أحد أعضائها.{{p}}ليس للنميمة أن تتسلل بينهم.{{p}}الكل يعمل بروح القوة لا الفشل.{{p}}الكل بروح الجدية لا يعرفون التراخي والكسل.{{p}}* أنت رصيد الكنيسة وغناها وحصنها.{{p}}اسمك حلو، واهب القوة والنصرة.
