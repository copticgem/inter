[٧]{{/b}}{{p}}{{b}}وَمِنَ الْجَادِيِّينَ انْفَصَلَ إِلَى دَاوُدَ إِلَى الْحِصْنِ فِي الْبَرِّيَّةِ {{/b}}{{p}}{{b}}جَبَابِرَةُ الْبَأْسِ رِجَالُ جَيْشٍ لِلْحَرْبِ، {{/b}}{{p}}{{b}}صَافُّو أَتْرَاسٍ وَرِمَاحٍ، {{/b}}{{p}}{{b}}وَوُجُوهُهُمْ كَوُجُوهِ الأُسُودِ، {{/b}}{{p}}{{b}}وَهُمْ كَالظَّبْيِ عَلَى الْجِبَالِ فِي السُّرْعَةِ: [٨]{{/b}}{{p}}{{b}}عَازَرُ الرَّأْسُ وَعُوبَدْيَا الثَّانِي وَأَلِيآبُ الثَّالِثُ [٩]{{/b}}{{p}}{{b}}وَمِشْمِنَّةُ الرَّابِعُ وَيَرْمِيَا الْخَامِسُ [١٠]{{/b}}{{p}}{{b}}وَعَتَّايُ السَّادِسُ وَإِيلِيئِيلُ السَّابِعُ [١١]{{/b}}{{p}}{{b}}وَيُوحَانَانُ الثَّامِنُ وَأَلْزَابَادُ التَّاسِعُ [١٢]{{/b}}{{p}}{{b}}وَيَرْمِيَا الْعَاشِرُ وَمَخْبَنَّايُ الْحَادِي عَشَرَ. [١٤]{{/b}}{{p}}{{b}}هَؤُلاَءِ هُمُ الَّذِينَ عَبَرُوا الأُرْدُنَّ فِي الشَّهْرِ الأَوَّلِ {{/b}}{{p}}{{b}}وَهُوَ مُمْتَلِئٌ إِلَى جَمِيعِ شُطُوطِهِ وَهَزَمُوا كُلَّ أَهْلِ الأَوْدِيَةِ شَرْقًا وَغَرْبًا. [١٥]{{/b}}{{p}}{{b}}وَجَاءَ قَوْمٌ مِنْ بَنِي بِنْيَامِينَ وَيَهُوذَا إِلَى الْحِصْنِ إِلَى دَاوُدَ. [١٦]{{/b}}{{p}}{{b}}فَخَرَجَ دَاوُدُ لاِسْتِقْبَالِهِمْ وَقَالَ لَهُمْ: {{/b}}{{p}}{{b}}«إِنْ كُنْتُمْ قَدْ جِئْتُمْ بِسَلاَمٍ إِلَيَّ لِتُسَاعِدُونِي، {{/b}}{{p}}{{b}}يَكُونُ لِي مَعَكُمْ قَلْبٌ وَاحِدٌ. {{/b}}{{p}}{{b}}وَإِنْ كَانَ لِتَدْفَعُونِي لِعَدُوِّي وَلاَ ظُلْمَ فِي يَدَيَّ، {{/b}}{{p}}{{b}}فَلْيَنْظُرْ إِلَهُ آبَائِنَا وَيُنْصِفْ». [١٧]{{/b}}{{p}}{{b}}فَحَلَّ الرُّوحُ عَلَى عَمَاسَايَ رَأْسِ الثَّوَالِثِ فَقَالَ:{{/b}}{{p}}{{b}}«لَكَ نَحْنُ يَا دَاوُدُ، {{/b}}{{p}}{{b}}وَمَعَكَ نَحْنُ يَا ابْنَ يَسَّى. {{/b}}{{p}}{{b}}سَلاَمٌ سَلاَمٌ لَكَ، وَسَلاَمٌ لِمُسَاعِدِيكَ. {{/b}}{{p}}{{b}}لأَنَّ إِلَهَكَ مُعِينُكَ». {{/b}}{{p}}{{b}}فَقَبِلَهُمْ دَاوُدُ وَجَعَلَهُمْ رُؤُوسَ الْجُيُوشِ. [١٨]{{/b}}{{p}}{{b}}وَسَقَطَ إِلَى دَاوُدَ بَعْضٌ مِنْ مَنَسَّى {{/b}}{{p}}{{b}}حِينَ جَاءَ مَعَ الْفِلِسْطِينِيِّينَ ضِدَّ شَاوُلَ لِلْقِتَالِ وَلَمْ يُسَاعِدُوهُمْ، {{/b}}{{p}}{{b}}لأَنَّ أَقْطَابَ الْفِلِسْطِينِيِّينَ أَرْسَلُوهُ بِمَشُورَةٍ قَائِلِينَ: {{/b}}{{p}}{{b}}«إِنَّمَا بِرُؤُوسِنَا يَسْقُطُ إِلَى سَيِّدِهِ شَاوُلَ». [١٩]{{/b}}{{p}}{{b}}حِينَ انْطَلَقَ إِلَى صِقْلَغَ {{/b}}{{p}}{{b}}سَقَطَ إِلَيْهِ مِنْ مَنَسَّى عَدْنَاحُ وَيُوزَابَادُ وَيَدِيعَئِيلُ وَمِيخَائِيلُ وَيُوزَابَادُ وَأَلِيهُو وَصِلْتَايُ رُؤُوسُ أُلُوفِ مَنَسَّى. [٢٠]{{/b}}{{p}}{{b}}وَهُمْ سَاعَدُوا دَاوُدَ عَلَى الْغُزَاةِ لأَنَّهُمْ جَمِيعًا جَبَابِرَةُ بَأْسٍ، {{/b}}{{p}}{{b}}وَكَانُوا رُؤَسَاءَ فِي الْجَيْشِ. [٢١]{{/b}}{{p}}{{b}}لأَنَّهُ وَقْتَئِذٍ أَتَى أُنَاسٌ إِلَى دَاوُدَ يَوْمًا فَيَوْمًا لِمُسَاعَدَتِهِ {{/b}}{{p}}{{b}}حَتَّى صَارُوا جَيْشًا عَظِيمًا كَجَيْشِ اللَّهِ. [٢٢]{{/b}}{{p}}{{b}}وَهَذَا عَدَدُ رُؤُوسِ الْمُتَجَرِّدِينَ لِلْقِتَالِ الَّذِينَ جَاءُوا إِلَى دَاوُدَ إِلَى حَبْرُونَ {{/b}}{{p}}{{b}}لِيُحَوِّلُوا مَمْلَكَةَ شَاوُلَ إِلَيْهِ حَسَبَ قَوْلِ الرَّبِّ. [٢٣]{{/b}}{{p}}{{b}}بَنُو يَهُوذَا حَامِلُو الأَتْرَاسِ وَالرِّمَاحِ سِتَّةُ آلاَفٍ {{/b}}{{p}}{{b}}وَثَمَانِ مِئَةِ مُتَجَرِّدٍ لِلْقِتَالِ. [٢٤]{{/b}}{{p}}{{b}}مِنْ بَنِي شَمْعُونَ جَبَابِرَةُ بَأْسٍ فِي الْحَرْبِ سَبْعَةُ آلاَفٍ وَمِئَةٌ. [٢٥]{{/b}}{{p}}{{b}}مِنْ بَنِي لاَوِي أَرْبَعَةُ آلاَفٍ وَسِتُّ مِئَةٍ. [٢٦]{{/b}}{{p}}{{b}}وَيَهُويَادَاعُ رَئِيسُ الْهَارُونِيِّينَ وَمَعَهُ ثَلاَثَةُ آلاَفٍ وَسَبْعُ مِئَةٍ. [٢٧]{{/b}}{{p}}{{b}}وَصَادُوقُ غُلاَمٌ جَبَّارُ بَأْسٍ وَبَيْتُ أَبِيهِ اثْنَانِ وَعِشْرُونَ قَائِدًا. [٢٨]{{/b}}{{p}}{{b}}وَمِنْ بَنِي بِنْيَامِينَ إِخْوَةُ شَاوُلَ ثَلاَثَةُ آلاَفٍ {{/b}}{{p}}{{b}}(وَإِلَى هُنَا كَانَ أَكْثَرُهُمْ يَحْرُسُونَ حِرَاسَةَ بَيْتِ شَاوُلَ). [٢٩]{{/b}}{{p}}{{b}}وَمِنْ بَنِي أَفْرَايِمَ عِشْرُونَ أَلْفًا وَثَمَانُ مِئَةٍ، {{/b}}{{p}}{{b}}جَبَابِرَةُ بَأْسٍ وَذَوُو اسْمٍ فِي بُيُوتِ آبَائِهِمْ. [٣٠]{{/b}}{{p}}{{b}}وَمِنْ نِصْفِ سِبْطِ مَنَسَّى ثَمَانِيَةُ عَشَرَ أَلْفًا {{/b}}{{p}}{{b}}قَدْ تَعَيَّنُوا بِأَسْمَائِهِمْ لِيَأْتُوا وَيُمَلِّكُوا دَاوُدَ. [٣١]{{/b}}{{p}}{{b}}وَمِنْ بَنِي يَسَّاكَرَ الْخَبِيرِينَ بِالأَوْقَاتِ لِمَعْرِفَةِ مَا يَعْمَلُ إِسْرَائِيلُ، {{/b}}{{p}}{{b}}رُؤُوسُهُمْ مِئَتَانِ وَكُلُّ إِخْوَتِهِمْ تَحْتَ أَمْرِهِمْ. [٣٢]{{/b}}{{p}}{{b}}مِنْ زَبُولُونَ الْخَارِجُونَ لِلْقِتَالِ الْمُصْطَفُّونَ لِلْحَرْبِ {{/b}}{{p}}{{b}}بِجَمِيعِ أَدَوَاتِ الْحَرْبِ خَمْسُونَ أَلْفًا، {{/b}}{{p}}{{b}}وَلِلاِصْطِفَافِ مِنْ دُونِ خِلاَفٍ. [٣٣]{{/b}}{{p}}{{b}}وَمِنْ نَفْتَالِي أَلْفُ رَئِيسٍ وَمَعَهُمْ سَبْعَةٌ وَثَلاَثُونَ أَلْفًا بِالأَتْرَاسِ وَالرِّمَاحِ. [٣٤]{{/b}}{{p}}{{b}}وَمِنَ الدَّانِيِّينَ مُصْطَفُّونَ لِلْحَرْبِ ثَمَانِيَةٌ وَعِشْرُونَ أَلْفًا وَسِتُّ مِئَةٍ. [٣٥]{{/b}}{{p}}{{b}}وَمِنْ أَشِيرَ الْخَارِجُونَ لِلْجَيْشِ لأَجْلِ الاِصْطِفَافِ لِلْحَرْبِ أَرْبَعُونَ أَلْفًا. [٣٦]{{/b}}{{p}}{{b}}وَمِنْ عَبْرِ الأُرْدُنِّ مِنَ الرَّأُوبَيْنِيِّينَ وَالْجَادِيِّينَ وَنِصْفِ سِبْطِ مَنَسَّى {{/b}}{{p}}{{b}}بِجَمِيعِ أَدَوَاتِ جَيْشِ الْحَرْبِ مِئَةٌ وَعِشْرُونَ أَلْفًا. [٣٧]{{/b}}{{p}}{{b}}كُلُّ هَؤُلاَءِ رِجَالُ حَرْبٍ يَصْطَفُّونَ صُفُوفًا، {{/b}}{{p}}{{b}}أَتُوا بِقَلْبٍ تَامٍّ إِلَى حَبْرُونَ لِيُمَلِّكُوا دَاوُدَ عَلَى كُلِّ إِسْرَائِيلَ. {{/b}}{{p}}{{b}}وَكَذَلِكَ كُلُّ بَقِيَّةِ إِسْرَائِيلَ بِقَلْبٍ وَاحِدٍ لِتَمْلِيكِ دَاوُدَ. القارئ أنه في فترة القضاة كان الفلسطينيون قد استولوا على تابوت العهد، ولكن لأنه سبب لهم متاعب لا حصر لها قاموا بوضعه على عجلة جديدة وأعادوه إلى إسرائيل (١صم ٦)، ومن ذك الوقت بقى تابوت العهد في منزل أبيناداب في قرية يعاريم، إلى أن رأى داود أن يقوم بمحاولة لإرجاع تابوت الله إلى عاصمته أورشليم، لكيما تصبح المدينة الملكية هي أيضًا المدينة المقدسة.{{p}}ويمكننا أن نتعلم الكثير من هذه الفترة لفائدتنا واللاويون في مدن مسارحهم ليجتمعوا إلينا فنرجع تابوت إلهنا إلينا لأننا لم نسأل به في أيام شاول" (١أيام ١٣: ١، ٣).{{p}}هذا الجزء من القصة لم يُذكر في سفر صموئيل، ونلاحظ:{{p}}١.أنه بمجرد أن تثبَّت داود على عرشه فكّر في تابوت الله وكان له هدفان: {{p}}أ. تقديم الكرامة لله، بإظهار الاحترام لتابوته الذي يمثل حضوره. يا ليت الاهتمام الأول والأعظم لأولئك الذين تمتعوا بالغنى والكرامة أن يكون تقديم الكرامة لله وخدمته وخدمة ملكوته بين البشر.{{p}}ب. الحصول على الراحة والفائدة من هذه الوساطة. إنه من الحكمة أن يأخذ حكام العالم تابوت الله معهم، ليتخذوا من حكمته مشورتهم ومن نواميسه حكمهم، فالذين يبدأون بمخافة الله يحظون بنعمته.{{p}}٢. أنه قام باستشارة قادة الشعب، فبالرغم من أنه كان عمًلا جليًلا، وبما أن داود هو الملك ومن سلطانه أن يأمر بتنفيذ هذا العمل، إلاَّ أنه اختار أن يأخذ المشورة. ليس من مُدبّر حكيم يشتهي أن يتخذوا قراراته دون مشورة. فداود نفسه كان ذكيًا جدًا، ومع ذلك استشار قواده، "لأن الخلاص بكثرة المشيرين"(أم ٢٤: ٦).{{p}}٣. أن قام باستدعاء كل الشعب ليحضروا هذه المناسبة، لأجل تكريم التابوت ولأجل إرضاء الشعب وتنويره، ونلاحظ أنه يدعوا عامة الشعب اخوة وهذا يدل على تواضعه وكياسته ومدى المحبة التي كان يكنها لهم، هكذا ربنا يسوع لم يستحي أن يدعونا اخوة (عب ٢: ١١).{{p}}٤. أنه اشترط أن يكون هذا حسب إرادة الله "بالرغم من أن هذا قد يظهر حسن لي ولكم، إلاَّ أنه لم يكن من الرب فسوف لا نعمله". ليتنا نحن أيضًا في كل أمر نفعله يجب أن يكون سؤالنا: "هل هذا من الله؟ هل هذا حسب إرادته؟ هل يمكننا أن نعطي جوابًا حسنًا أمام الله عن هذا؟ هل نتوقع أن يكون هذا مرضيًا لله؟{{p}}٥. أنه من المتطّلب أن يصلحوا ما قد أُهمل أثنا المُلك السابق، وأن يكفّروا عن اهمالهم. ونلاحظ أن داود لم يبدِ أيّ ملاحظة سيئة عن شاول، فلم يقل "شاول لم يهتم أبدًا بالتابوت، على الأقل في أواخر مُلكْه" ولكن قال باتضاع "لأننا لم نسأل به في أيام شاول" فاعتبرنفسع مذنبًا مع الآخرين عن هذا الاهمال. أنه من الأفضل أن نحكم على أنفسنا وليس على الآخرين، فالمتضعون والصالحون "يندبون نصيبهم في الذنب العام ويأخذون العار لأنفسهم" (دا ٩: ٥ وما يليه).{{p}}*     "واركبوا تابوت الله على عجلة جديدة من بيت أبيناداب وكان عُزّا وأخيو يسوقان العجلة..."(١أيام ١٣: ٧ وما يليه).{{p}} ربما يشير كاتب الأخبار بهذه الحقبة ليشدد على أهمية اللاويين الذين كان من الواجب أن يكلفوا لنقل التابوت (انظر ١٥: ١١، ١٥).{{p}}إن عجلة يجرها ثيران كانت وسيلة مناسبة للفلسطينين لنقل التابوت (انظر ١صم ٦: ٧)، ولكن شعب الله قد أعطى أوامر محددة في هذا المضمار (انظر عدد ٤: ١١، ١٥)، فبالرغم من أنهم احتفلوا بحضور التابوت إلا أنهم عصوا تعليمات الله وبهذا فقد فشلوا في الاعتراف بوجود الله القدوس بينهم. من كان يظن أن داود كان من الممكن أن يعمل مثل هذه الغلطة بحمل التابوت على عجلة؟ لأن الفلسطينيين حملوه بهذه الطريقة والتدبير الإلهي سمح بدفع العجلة، ظن داود أنه من الممكن أن يعملوا المِثْل. ولكن يجب أن نسير طبقًا للقاعدة وليس طبقًا للتمثل حينما يختلف عن القاعدة، حتى وأن كانت مثل هذه الأمثال قد سمحت بها العناية الإلهية في ظروف خاصة. ولذلك لما لمس عُزّا التابوت ضربه الرب ومات، وخطية عُزّا تنذرنا بقوة لنتجنب الأفتراض المُسبّق والتسرع وعدم الهيبة أثناء التعامل مع المقدسات، ولا نظن أن حسن النية يبرر العمل الخاطئ. فمثًلا عند التقدم من الأسرار المقدسة يجب أن نلاحظ ونفحص قلوبنا، لئلا تُؤدي الزلة والألفة والاعتياد إلى قلة الاحترام. ليتنا لا نعبث في طرقنا للأقتراب من الله، ولكن ليتنا نتقدم خلال المسيح بدالة إلى عرش النعمة، لأننا تحت تدبير الحرية والنعمة وليس تحت العبودية والارهاب، ليت الكآبة التي بدّلت فرح إسرائيل بسبب عقوبة عُزّا تذكرنا دائمًا أن نعبد الرب بخوف ورعدة بينما نخدم بفرح وسرور، ليكن غيظ داود في هذه المناسبة محذرًا لنا لنلاحظ أنفسنا حينما نكون تحت تأديب من الله لئلا نتخاصم معه بدًلا من الخضوع لارادته، ولتكن البركة التي حلّت في بيت عوبيد أدوم بسبب التابوت مشّجعة لنا لنرحب بطقوس الكنيسة في بيوتنا، أيّ كمن يؤمنون أن التابوت كضيف لا يسبب ضررًا، وإنما كانت كلمة الإنجيل للبعض هي "رائحة موت لموت" كما كان التابوت لعُزّا، فلنقبلها نحن بفرح ومحبة وسوف تكون لنا "رائحة حياة لحياة".  {{p}}{{b}}
