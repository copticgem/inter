{{p}}{{v1}}{{b}} آية ١: "لِتَثْبُتِ الْمَحَبَّةُ الأَخَوِيَّةُ."{{/b}}{{p}}{{b}} لتثبت المحبة الاخوية{{/b}} = الرسول كعادته يختم رسالته بحديث عملي. فهو في ص١٢ حدثهم عن الجهاد وهنا يحدثهم عن جوانب عملية مثل المحبة والتسبيح والطاعة. وقوله هنا لتثبت المحبة يشير أن لهم أعمال محبة ويطلب منهم أن تثبت. فهذه كلمة تشجيع (١تس٩:٤). ولننعم بعمل المسيح الكفاري يلزمنا أن نعلن محبتنا للآخرين. فكلما اتسع قلبنا خلال عمل الله ومحبته، أحببنا نحن إخوتنا. وكلما أحببنا إخوتنا أعلن الله بالأكثر حبه فينا.{{p}}{{v2}}{{b}} آية ٢: "لاَ تَنْسَوْا إِضَافَةَ الْغُرَبَاءِ، لأَنْ بِهَا أَضَافَ أُنَاسٌ مَلاَئِكَةً وَهُمْ لاَ يَدْرُونَ."{{/b}}{{p}}كغرباء في العالم علينا أن نهتم بالغرباء وكمتضايقين علينا أن نسند المتضايقين بالحب العامل لا بالكلام فقط. ولنلاحظ أن محبة الغرباء وإضافتهم هي محك صدق لطبيعة المحبة التي ذكرها في آية (١). {{b}}أناس{{/b}} = أبراهيم ولوط.{{p}}{{v3}}{{b}} آية ٣: "اُذْكُرُوا الْمُقَيَّدِينَ كَأَنَّكُمْ مُقَيَّدُونَ مَعَهُمْ، وَالْمُذَلِّينَ كَأَنَّكُمْ أَنْتُمْ أَيْضًا فِي الْجَسَدِ."{{/b}}{{p}}قارن مع (مت٢٥: ٣٥، ٣٦).{{p}}قال {{b}}أذكروا المقيدين كأنكم مقيدون{{/b}} فكان في المقابل يجب أن يقول والمذلين كأنكم مذلولين. ولكنه قال كأنكم أنتم أيضا في الجسد. فمن هو في الجسد يشعر في وقت الإضطهاد أنه مذلول فهو محروم من راحة جسده وشهوات جسده لكن من هو في الروح {{b}}أولاً:{{/b}} لن يضيره شيئا من حرمانه من الملذات الجسدية والراحة. {{b}}ثانيا:{{/b}} هو سيشعر أنه شريك آلام وشريك مجد مع المسيح. وكل ما يسمح به المسيح هو خير وعلامة محبة، فمن هو في الروح لن يشعر أبدا بالذل فهو في حرية مجد أولاد الله (رو٨: ٢١) + "يتجسسوا حريتنا التي لنا في المسيح كي يستعبدونا" (غل٢: ٤). فمن هم في المسيح هم في حرية. أما من هم في الجسد فمع كل ألم يشعرون أنهم في ذل. {{p}}{{v4}}{{b}} آية ٤: "لِيَكُنِ الزِّوَاجُ مُكَرَّمًا عِنْدَ كُلِّ وَاحِدٍ، وَالْمَضْجَعُ غَيْرَ نَجِسٍ. وَأَمَّا الْعَاهِرُونَ وَالزُّنَاةُ فَسَيَدِينُهُمُ اللهُ."{{/b}}{{p}}الرسول يكرم الزواج لسببين:{{p}} ١. خوفًا من أن البتوليون لا يكرمونه.{{p}} ٢. إشارة لبعض الهرطقات التي علمت أن الزواج نجس.{{p}}هذا فضلًا عن أنه في أيام نهاية اليهودية انتشرت العادات البطالة والزنا والطلاق والرسول ينبه أي يحذر أن نكون مثلهم لا نكرم الزواج. {{b}}الزناة{{/b}} = تشير للخيانة الزوجية. {{b}}العاهرون{{/b}} = من يمارس عادات أو ممارسات شاذة محرمة.{{p}}{{v5}}{{b}} آية ٥: "لِتَكُنْ سِيرَتُكُمْ خَالِيَةً مِنْ مَحَبَّةِ الْمَالِ. كُونُوا مُكْتَفِينَ بِمَا عِنْدَكُمْ، لأَنَّهُ قَالَ:«لاَ أُهْمِلُكَ وَلاَ أَتْرُكُكَ»."{{/b}}{{p}}المال هو السيد الآخر الذي لا يرحم بل يستعبد. فعلينا بالقناعة. {{p}}{{b}}«لاَ أُهْمِلُكَ وَلاَ أَتْرُكُكَ» = {{/b}}(تث٣١: ٦، ٨ + يش١: ٥).{{p}}{{v6}}{{b}} آية ٦: "حَتَّى إِنَّنَا نَقُولُ وَاثِقِينَ:«الرَّبُّ مُعِينٌ لِي فَلاَ أَخَافُ. مَاذَا يَصْنَعُ بِي إِنْسَانٌ؟»"{{/b}}{{p}}من يصدق وعد الله "{{b}}لا أهملك ولا أتركك{{/b}}" ويتخلى عن الطمع سيختبر يد الله ومعونتهفيردد بثقة{{b}} :«لاَ أُهْمِلُكَ وَلاَ أَتْرُكُكَ»." وَاثِقِينَ{{/b}} = الفعل يعنى حالة الفرح مع الثقة.{{p}}البداية هي تصديق وعد الله وإتخاذ قرار بعدم الطمع والله مسئول عن نمو الإيمان وإزدياد خبرات الإنسان مع الله، وتزداد الثقة.{{p}}{{b}}لاَ أَخَافُ. مَاذَا يَصْنَعُ بِي إِنْسَانٌ{{/b}} = على المؤمن أن يثق في الله فلا سلطان لإنسان أن يؤذيه وطبعاً لا سلطان لفقر أو عوز أو أي ضيقة أن تطوله إن ظل متمسكاً بالرب. {{p}}{{v7}}{{b}} آية ٧: "اُذْكُرُوا مُرْشِدِيكُمُ الَّذِينَ كَلَّمُوكُمْ بِكَلِمَةِ اللهِ. انْظُرُوا إِلَى نِهَايَةِ سِيرَتِهِمْ فَتَمَثَّلُوا بِإِيمَانِهِمْ."{{/b}}{{p}}{{b}} مرشديكم{{/b}}= هم الرعاة الذين يكرزون بكلمة الله. والآباء الذين يجب أن نتمسك بإيمانهم المسلم لنا (أهمية التقليد في الكنيسة) وهذا التقليد هو سر بقاء وحدة الكنيسة حتى الآن بفكر واحد.{{p}}{{v8}}{{b}} آية ٨: "يَسُوعُ الْمَسِيحُ هُوَ هُوَ أَمْسًا وَالْيَوْمَ وَإِلَى الأَبَدِ."{{/b}}{{p}}إذ أراد أن يوصيهم في آية (٩) أن لا ينساقوا وراء تعاليم غريبة، يؤكد لهم هنا أن يسوع المسيح لا يتغير، نقبله كما قبله آباؤنا. ونسلم الإيمان الذي قبلناه من أبائنا إلى أولادنا كما هو (يه٣) ويستمر للأجيال القادمة بلا تحريف. المسيح عمل في أبائنا ويعمل فينا وسيعمل في الأجيال القادمة بنفس طريقته فهو لا يتغير كما كان مع الآباء هكذا سيكون معنا. لقد اختبرنا أعمال محبته ورحمته دائمًا فلماذا ينسانا اليوم ويهملنا.{{p}}{{v9}}{{b}} آية ٩: "لاَ تُسَاقُوا بِتَعَالِيمَ مُتَنَوِّعَةٍ وَغَرِيبَةٍ، لأَنَّهُ حَسَنٌ أَنْ يُثَبَّتَ الْقَلْبُ بِالنِّعْمَةِ، لاَ بِأَطْعِمَةٍ لَمْ يَنْتَفِعْ بِهَا الَّذِينَ تَعَاطَوْهَا."{{/b}}{{p}}يلمح الرسول هنا إلى هرطقات بدأت تنتشر في أيامه وهي تعلم أن الزواج نجس وأن هناك أطعمة نجسة يلزم الامتناع عنها. وكان الكهنة يأكلون لحوم ذبائح أوثانهم ليتقدسوا بها. أما اليهود فكان كهنتهم يأكلون بعض أنواع لحوم الذبائح ولا يأكلون من بعضها. فذبيحة الكفارة لأنها حاملة لخطايا الشعب لا يأكلون منها. وهنا الرسول يوصى بأن المسيحية ليس لها كل هذه الأفكار فلا يوجد طعام يسمى نجسًا والزواج مكرم. ولا يوجد طعام يقدس إلا جسد المسيح ودمه (لذلك في آية ١٠ مباشرة يذكر موضوع التناول) (تى٣:٤-٥).{{p}}{{v10}}{{b}} آية ١٠: "لَنَا «مَذْبَحٌ» لاَ سُلْطَانَ لِلَّذِينَ يَخْدِمُونَ الْمَسْكَنَ أَنْ يَأْكُلُوا مِنْهُ."{{/b}}{{p}}هنا يرفع الرسول فكرهم من التفكير في هل الطعام نجس أو غير نجس، هل يقدس أو لا يقدس، إلى ما أعطاه لنا المسيح وهو شركة الإفخارستيا. وهذا الكلام موجه للعبرانيين الذين بدأوا يشعرون أنهم محرومون من الهيكل.{{p}}{{b}} لنا مذبح{{/b}} = المسيح قدم نفسه على الصليب، قدم عليه كذبيحة كفارة عظمى حملت كل خطايا العالم. واستمرارًا لهذه الذبيحة هيأ لنا المسيح مائدة مقدسة من جسده ودمه.{{p}}{{b}}لاَ سُلْطَانَ {{/b}} = كهنة اليهود لا سلطان لهم أن يتناولوا من مائدة الإفخارستيا أولاً لأنهم غير مؤمنين وثانياً لأنه بحسب ناموسهم فهم لا سلطان لهم أن يتناولوا أو يأكلوا من ذبيحة الكفارة والمسيح ذبيحة كفارة عنا وحامل خطايانا. ورئيس الكهنة لا يأكل من ذبيحة الخطية لو كان يقدمها عن خطيته أو خطية كل الجماعة. والتناول هو ذبيحة خطية عن كل العالم وعن الكهنة ورؤساء الكهنة، وبحسب ناموس اليهود لا يحل لكهنة اليهود الأكل منها.{{p}}{{v11}}{{b}} آية ١١: "فَإِنَّ الْحَيَوَانَاتِ الَّتِي يُدْخَلُ بِدَمِهَا عَنِ الْخَطِيَّةِ إِلَى «الأَقْدَاسِ» بِيَدِ رَئِيسِ الْكَهَنَةِ تُحْرَقُ أَجْسَامُهَا خَارِجَ الْمَحَلَّةِ."{{/b}}{{p}}والدليل على أن كهنة الناموس اليهودى ليس لهم سلطان على أن يأكلوا من هذه الذبيحة الإفخارستية أنهم ما كانوا يأكلون من ذبيحة يوم الكفارة، هذه التي يدخل بدمها عن الخطية بيد رئيس الكهنة وتحرق أجسامها خارج المحلة (لا١٦: ٢٧) وكان رئيس الكهنة يوم الكفارة يأخذ من دم هذه الذبيحة ويدخل به للأقداس أما الذبيحة نفسها فلا يأكل منها بل تحرق أجسامها خارج المحلة بحسب الناموس وذبيحة الكفارة هذه هي التي تشير لذبيحة الصليب وبالتالي للإفخارستيا إمتداد ذبيحة الصليب. فلا سلطان لهم أن يأكلوا منها بحسب ناموسهم.{{p}}{{b}}لاحظ{{/b}} أن شريعة ذبيحة الخطية " الكاهن الذي يعملها للخطية يأكلها " (لا٦: ٢٦)، وذلك لأن الكاهن هنا بكهنوته يمثل المسيح رئيس كهنتنا الذي حمل خطيتنا ومات بها فأماتها، وكرمز لذلك يأكل الكاهن من لحم ذبيحة الخطية، وكأنه بأكله من الذبيحة التي حملت خطية الخاطئ قد شفع في الخاطئ وإبتلع خطيته، فإختفت إلى غير رجعة، فيذهب الخاطئ مقدم الذبيحة إلى بيته هادئ البال إذ غفرت خطيته. وهذا الطقس يشرح معنى شفاعة المسيح الكفارية في الخطاة، وهنا فالله يشرح المعنى لشعبه عن طريق الكاهن وكأن الكاهن يشفع في الخاطئ فترفع خطيته.{{p}}ولكن لو أخطأ رئيس الكهنة نفسه، كان لا يأكل من لحم ذبيحة الخطية، بل تحرق كلها{{p}}بالنار، فكيف يشفع في نفسه وهو نفسه خاطئ. لكن كان يدخل بدم الذبيحة إلى خيمة الإجتماع وينضح من الدم طالبا الغفران من الله (لا٤: ٥ – ٧). وراجع (لا١٠: ١٨-٢٠) لترى كيف فهم هرون هذا الطقس.{{p}}وبنفس الطريقة نجد أنه لو أخطأت كل الجماعة كان رئيس الكهنة يفعل نفس الشئ ويحرق كل الذبيحة، وكأنه هو المسئول، فكيف يشفع في الشعب وهو المسئول عن فساد الشعب.{{p}}وقد لخص بولس الرسول هذا الطقس فقال " فإن الْحَيَوَانَاتِ الَّتِي يُدْخَلُ بِدَمِهَا عَنِ الْخَطِيَّةِ إِلَى «الأَقْدَاسِ» بِيَدِ رَئِيسِ الْكَهَنَةِ تُحْرَقُ أَجْسَامُهَا خَارِجَ الْمَحَلَّةِ " (عب١٣: ١١). فهنا نجد رئيس الكهنة اليهودى غير قادر على أن يشفع في نفسه أو في شعبه إذ أنه هو نفسه خاطئ، فلا يأكل من لحم الذبيحة بل يحرقه. ويذهب بدم الذبيحة إلى داخل الخيمة طالبا الغفران من الله لنفسه وعن الشعب.{{p}}أما المسيح الذي بلا خطية وحده فلقد قدم نفسه كرئيس كهنتنا، وأكلته نيران العدالة الإلهية، بل قل نيران المحبة الإلهية ليرفع خطاينا ويدخل بدمه إلى الأقداس السمائية لا ليطلب غفرانا لنفسه بل ليشفع فينا شفاعة أبدية " وليس بدم تيوس وعجول بل بدم نفسه دخل مرة واحدة إلى الأقداس فوجد فداءً أبديا " (عب٩: ١٢). {{p}}{{v12}}{{b}} آية ١٢: "لِذلِكَ يَسُوعُ أَيْضًا، لِكَيْ يُقَدِّسَ الشَّعْبَ بِدَمِ نَفْسِهِ، تَأَلَّمَ خَارِجَ الْبَابِ."{{/b}}{{p}}هنا يفسر الرسول الرمز في حرق أجسام ذبيحة الكفارة خارج أورشليم وأن هذا كان رمزًا لصلب المسيح خارج أورشليم. ولكن المسيح لم يحرق بنار ولكن نار اللاهوت التي فيه أحرقت الخطايا والتهمتها. هذا هو روح الإحراق وروح التطهير (إش٤:٤).  وخروجه وصلبه خارج أورشليم أفاد هجرانه للأمه اليهودية وهيكلها لذلك قال لهم المسيح ها بيتكم يترك لكم خرابًا (مت٢٣: ٣٨). هم طردوا ملكهم فصاروا بلا ملك للأبد.{{p}}{{v13}}{{b}} آية ١٣: "فَلْنَخْرُجْ إِذًا إِلَيْهِ خَارِجَ الْمَحَلَّةِ حَامِلِينَ عَارَهُ."{{/b}}{{p}}إن كان مسيحنا قد طردوه وصلبوه فلا نخجل إن طردونا. ولنكن مستعدين أن نتقبل التعييرات لأجل اسم المسيح.{{p}}{{v14}}{{b}} آية ١٤: "لأَنْ لَيْسَ لَنَا هُنَا مَدِينَةٌ بَاقِيَةٌ، لكِنَّنَا نَطْلُبُ الْعَتِيدَةَ."{{/b}}{{p}}بعد أن خرجنا معه حاملين عاره نطلب أورشليم السماوية العتيدة. علينا دائمًا أن نترك عالم الخطية فالعالم ليس له صفة الدوام بل نتطلع لأورشليم السماوية.{{p}}{{v15}}{{b}} آية ١٥: "فَلْنُقَدِّمْ بِهِ فِي كُلِّ حِينٍ ِللهِ ذَبِيحَةَ التَّسْبِيحِ، أَيْ ثَمَرَ شِفَاهٍ مُعْتَرِفَةٍ بِاسْمِهِ."{{/b}}{{p}}الطرد أو الإهانة لا يجب أن تنشئ تبرمًا أو ضيقًا بل نظل على الدوام مسبحين ولنذكر أن التلاميذ إذ ضربوهم وأهانوهم ذهبوا فرحين.. لأنهم حسبوا مستأهلين أن يهانوا من أجل اسمه (أع٥: ٤١). فالإهانة هنا هي شركة ألم مع المسيح. {{b}}فلنقدم به{{/b}} = لنرفع به أو بواسطته لله {{b}}ذبيحة التسبيح{{/b}} = (هذه هي ذبائح المسيحيين) فلا قدوم للآب سوى به، فبه نتقدم وبدونه لا نقدر أن نفعل شيء (يو١٥: ٥). نحن نقدم له ذبيحتنا وهو كرئيس كهنة يقدمها على مذبح الله (١بط٢: ٥). وراجع (رو١: ٨) فحتى الشكر لا نقوى عليه سوى بيسوع المسيح (نحن غير مقبولين إلا به أي بيسوع المسيح) + (كو٣: ١٧). إذًا النصيحة التي يقدمها الرسول للعبرانيين أن ينفصلوا عن الكهنوت اللاوي مقدمين ذبائح تسبيح.{{p}}{{v16}}{{b}} آية ١٦: "وَلكِنْ لاَ تَنْسَوْا فِعْلَ الْخَيْرِ وَالتَّوْزِيعَ، لأَنَّهُ بِذَبَائِحَ مِثْلِ هذِهِ يُسَرُّ اللهُ."{{/b}}{{p}}التسبيح ليس مجرد كلمات بل أفعال عملية تعبر عن طبيعة المؤمن ومشاعره المملوءة محبة. لذلك يطلب منهم {{b}}فعل الخير والتوزيع{{/b}} = كانوا يذبحون الذبائح ويوزعون على المحتاجين. أو يقيموا لهم وليمة في الكنيسة.{{p}}{{v17}}{{b}} آية ١٧: "أَطِيعُوا مُرْشِدِيكُمْ وَاخْضَعُوا، لأَنَّهُمْ يَسْهَرُونَ لأَجْلِ نُفُوسِكُمْ كَأَنَّهُمْ سَوْفَ يُعْطُونَ حِسَابًا، لِكَيْ يَفْعَلُوا ذلِكَ بِفَرَحٍ، لاَ آنِّينَ، لأَنَّ هذَا غَيْرُ نَافِعٍ لَكُمْ.{{/b}}{{p}}{{b}} أطيعوا مرشديكم{{/b}} = لنلاحظ أن عدم احترام الرئاسة يسبب مصاعب جمة وشغب وفساد للشعب. {{b}}يسهرون لأجل نفوسكم{{/b}} = خطورة وظيفة المرشد أنه مسئول عن نفوس شعبه وخلاصهم. وسيقدم هذا المرشد حسابًا أمام الله عنهم. {{b}}لا آنين{{/b}} = هذا توبيخ للشعب لأنهم جعلوا مرشديهم يئنون من عنادهم أو انحلالهم أو عدم خضوعهم. لأن هذا هو غير نافع لكم = بسبب أنين المرشدين سيعاقبكم الله. {{b}}لكي يفعلوا هذا بفرح{{/b}} = أي يقوموا بخدمتكم وإرشادكم وهم فرحين بثمار خدمتهم.{{p}}{{v18}}{{b}} آية ١٨: "صَلُّوا لأَجْلِنَا، لأَنَّنَا نَثِقُ أَنَّ لَنَا ضَمِيرًا صَالِحًا، رَاغِبِينَ أَنْ نَتَصَرَّفَ حَسَنًا فِي كُلِّ شَيْءٍ."{{/b}}{{p}}الرئاسة في الكهنوت لا تعني الاستبداد والارستقراطية. فبولس الرئيس يطلب صلواتهم. هو لا يشعر بأفضلية عن الشعب بل يشعر باحتياجه لصلواتهم. {{b}}لنا ضمير صالحًا{{/b}} = فالصلاة للآخرين لن تنفعهم إن لم يكن لهم ضمير صالح.{{p}}{{v19}}{{b}} آية ١٩: "وَلكِنْ أَطْلُبُ أَكْثَرَ أَنْ تَفْعَلُوا هذَا لِكَيْ أُرَدَّ إِلَيْكُمْ بِأَكْثَرِ سُرْعَةٍ."{{/b}}{{p}}نلمح في هذه الآية أسلوب بولس الرسول. (فل٢٢) + (رو٢٢:١٥-٢٤) + (١كو٥:١٦-٧).{{p}}{{v20}}{{v21}}{{b}} آية ٢٠، ٢١: "وَإِلهُ السَّلاَمِ الَّذِي أَقَامَ مِنَ الأَمْوَاتِ رَاعِيَ الْخِرَافِ الْعَظِيمَ، رَبَّنَا يَسُوعَ، بِدَمِ الْعَهْدِ الأَبَدِيِّ، ٢١لِيُكَمِّلْكُمْ فِي كُلِّ عَمَل صَالِحٍ لِتَصْنَعُوا مَشِيئَتَهُ، عَامِلًا فِيكُمْ مَا يُرْضِي أَمَامَهُ بِيَسُوعَ الْمَسِيحِ، الَّذِي لَهُ الْمَجْدُ إِلَى أَبَدِ الآبِدِينَ. آمِينَ."{{/b}}{{p}}{{b}} إله السلام{{/b}} = الرسول ينسب صفة السلام لله فهو يكلم أناسًا فقدوا سلامهم بسبب الضيق. فكأنه يطلب السلام لهم من إله السلام. ولنلاحظ أن كلمات البركة في نهاية الرسالة متفقة مع موضوع الرسالة. {{b}}راعي الخراف{{/b}} = الشعب هو خرافه التي مات لأجلها وقام فهل يتركهم. هنا كان بولس يستعطف الله أن لا يهمل شعبه حتى وأن أهملوا وفكروا في الارتداد، بل يحميهم من الارتداد. قارن مع (١تس٥: ٢٣).{{p}}{{v22}}{{b}} آية ٢٢: "وَأَطْلُبُ إِلَيْكُمْ أَيُّهَا الإِخْوَةُ أَنْ تَحْتَمِلُوا كَلِمَةَ الْوَعْظِ، لأَنِّي بِكَلِمَاتٍ قَلِيلَةٍ كَتَبْتُ إِلَيْكُمْ."{{/b}}{{p}}أنا لم أشأ أن أتعبكم فكتبت كلمات قليلة، ياليتكم تحتملوها.{{p}}{{v23}}{{b}} آية ٢٣: "اِعْلَمُوا أَنَّهُ قَدْ أُطْلِقَ الأَخُ تِيمُوثَاوُسُ، الَّذِي مَعَهُ سَوْفَ أَرَاكُمْ، إِنْ أَتَى سَرِيعًا."{{/b}}{{p}}ربما تيموثاوس كان محبوسًا من اليهود أو غيرهم ثم أطلقوه وربما كان تيموثاوس محبوسًا مع بولس في روما وكان هذا الحبس سببًا في تأخير زيارة بولس لهم. وهذه من الآيات التي تشير أن بولس هو كاتب الرسالة. بسبب علاقته مع تيموثاوس وبسبب الأسلوب (١تس٢:٣).{{p}}{{v24}}{{b}} آية ٢٤: "سَلِّمُوا عَلَى جَمِيعِ مُرْشِدِيكُمْ وَجَمِيعِ الْقِدِّيسِينَ. يُسَلِّمُ عَلَيْكُمُ الَّذِينَ مِنْ إِيطَالِيَا."{{/b}}{{p}}{{v25}}{{b}} آية ٢٥: "اَلنِّعْمَةُ مَعَ جَمِيعِكُمْ. آمِينَ.{{/b}}{{p}}{{b}} إلى العبرانيين كتبت من إيطاليا على يد تيموثاوس."{{/b}}{{p}}{{b}}
