{{p}}{{v1}}{{b}}أية (١):- "١وَكَلَّمَ الرَّبُّ مُوسَى قَائِلاً: "{{/b}}{{p}}{{v2}}{{b}}أية (٢):- "{{/b}}{{b}}٢«كَلِّمْ هَارُونَ وَبَنِيهِ أَنْ يَتَوَقَّوْا أَقْدَاسَ بَنِي إِسْرَائِيلَ الَّتِي يُقَدِّسُونَهَا لِي وَلاَ يُدَنِّسُوا اسْمِي الْقُدُّوسَ. أَنَا الرَّبُّ. "{{/b}}{{p}}{{b}}يتوقوا أقداس{{/b}} = الأقداس هي الأنصبة المعينة للكهنة من الذبائح. {{b}}ويتوقوا{{/b}} أي لا يقتحموها أي يأكلوها بلا إستعداد، كأن تكون عليهم أي نجاسة. فهذه الأقداس ليست لإشباع بطونهم بل هي عمل مقدس يلزم ممارسته بفكر روحي. فحين يعترف الخاطئ على رأس الذبيحة بخطيته تنتقل الخطية للذبيحة. وحين يأكلها المذبح (النار) والكاهن. يشير هذا لأن الخطية قد إبتلعت وإنتهت، إشارة للمسيح الذي رفع خطايانا والآب الذي قبل هذا العمل. فنار المذبح تشير لقبول الآب للذبيحة، وأكل الكاهن لجزء منها يشير لحمل المسيح لخطيتى ثم مات بها فأماتها . وقوله {{b}}أنا الرب{{/b}} = أنا أغير على إسمى ومقدساتى التي يدنسها الكهنة المستهترين.{{p}}{{v3}}{{b}}أية (٣):- "{{/b}}{{b}}٣قُلْ لَهُمْ: فِي أَجْيَالِكُمْ كُلُّ إِنْسَانٍ مِنْ جَمِيعِ نَسْلِكُمُ اقْتَرَبَ إِلَى الأَقْدَاسِ الَّتِي يُقَدِّسُهَا بَنُو إِسْرَائِيلَ لِلرَّبِّ، وَنَجَاسَتُهُ عَلَيْهِ، تُقْطَعُ تِلْكَ النَّفْسُ مِنْ أَمَامِي. أَنَا الرَّبُّ."{{/b}}{{p}}{{b}}تقطع تلك النفس من أمامى{{/b}} = أي لا يعود للخدمة. {{b}}نجاسته عليه{{/b}} = لو أصيب الكاهن ببرص أو بسيل أو مس أحداً مريض بهما أو مس ميتاً. أو مس حيوان نجس أو إضطجع مع زوجته . هذا يبقى اليوم كله نجساً حتى المساء، ثم يرحض جسده بماء قبل أن يأكل من الأقداس.{{p}}{{v4}}{{v5}}{{v6}}{{v7}}{{v8}}{{v9}}{{b}}الأيات (٤-٩):-"{{/b}}{{b}}٤كُلُّ إِنْسَانٍ مِنْ نَسْلِ هَارُونَ وَهُوَ أَبْرَصُ أَوْ ذُو سَيْل، لاَ يَأْكُلْ مِنَ الأَقْدَاسِ حَتَّى يَطْهُرَ. وَمَنْ مَسَّ شَيْئًا نَجِسًا لِمَيْتٍ، أَوْ إِنْسَانٌ حَدَثَ مِنْهُ اضْطِجَاعُ زَرْعٍ، ٥أَوْ إِنْسَانٌ مَسَّ دَبِيبًا يَتَنَجَّسُ بِهِ، أَوْ إِنْسَانًا يَتَنَجَّسُ بِهِ لِنَجَاسَةٍ فِيهِ، ٦فَالَّذِي يَمَسُّ ذلِكَ يَكُونُ نَجِسًا إِلَى الْمَسَاءِ، وَلاَ يَأْكُلْ مِنَ الأَقْدَاسِ، بَلْ يَرْحَضُ جَسَدَهُ بِمَاءٍ. ٧فَمَتَى غَرَبَتِ الشَّمْسُ يَكُونُ طَاهِرًا، ثُمَّ يَأْكُلُ مِنَ الأَقْدَاسِ لأَنَّهَا طَعَامُهُ. ٨مِيْتَةً أَوْ فَرِيسَةً لاَ يَأْكُلْ فَيَتَنَجَّسَ بِهَا. أَنَا الرَّبُّ. ٩فَيَحْفَظُونَ شَعَائِرِي لِكَيْ لاَ يَحْمِلُوا لأَجْلِهَا خَطِيَّةً يَمُوتُونَ بِهَا لأَنَّهُمْ يُدَنِّسُونَهَا. أَنَا الرَّبُّ مُقَدِّسُهُمْ." {{/b}}{{p}}يحدد فيها ما ينجس الكاهن وكيف يتطهر. والكنيسة تلزم أبنائها أن لا يتقدموا للتناول دون إستحقاق أي يتقدموا وهم تائبين (١كو١١: ٢٩). {{p}}{{v10}}{{v11}}{{v12}}{{v13}}{{v14}}{{v15}}{{b}}الأيات (١٠-١٥):-"{{/b}}{{b}}١٠«وَكُلُّ أَجْنَبِيٍّ لاَ يَأْكُلُ قُدْسًا. نَزِيلُ كَاهِنٍ وَأَجِيرُهُ لاَ يَأْكُلُونَ قُدْسًا. ١١لكِنْ إِذَا اشْتَرَى كَاهِنٌ أَحَدًا شِرَاءَ فِضَّةٍ، فَهُوَ يَأْكُلُ مِنْهُ، وَالْمَوْلُودُ فِي بَيْتِهِ. هُمَا يَأْكُلاَنِ مِنْ طَعَامِهِ. ١٢وَإِذَا صَارَتِ ابْنَةُ كَاهِنٍ لِرَجُل أَجْنَبِيٍّ لاَ تَأْكُلُ مِنْ رَفِيعَةِ الأَقْدَاسِ. ١٣وَأَمَّا ابْنَةُ كَاهِنٍ قَدْ صَارَتْ أَرْمَلَةً أَوْ مُطَلَّقَةً، وَلَمْ يَكُنْ لَهَا نَسْلٌ، وَرَجَعَتْ إِلَى بَيْتِ أَبِيهَا كَمَا فِي صِبَاهَا، فَتَأْكُلُ مِنْ طَعَامِ أَبِيهَا. لكِنَّ كُلَّ أَجْنَبِيٍّ لاَ يَأْكُلُ مِنْهُ. ١٤وَإِذَا أَكَلَ إِنْسَانٌ قُدْسًا سَهْوًا، يَزِيدُ عَلَيْهِ خُمْسَهُ وَيَدْفَعُ الْقُدْسَ لِلْكَاهِنِ. ١٥فَلاَ يُدَنِّسُونَ أَقْدَاسَ بَنِي إِسْرَائِيلَ الَّتِي يَرْفَعُونَهَا لِلرَّبِّ، "{{/b}}{{p}}يتمتع بأكل الذبيحة الكاهن ومولود بيته ومن إشتراه الكاهن بفضة ولا يأكل معه في هذه المقدسات أجنبى أي عبرانى ليس من نسل هرون (خر٢٩: ٣٣) أو من كان غريب الجنس أو عبداً ثقبت أذنه، أو من كان نزيلاً (ضيف أو أجير أو عبد عبرانى فالعبد العبرانى لا يسمى عبد). ولا يأكل منها إبنته التي تزوجت بمن ليس من نسل هرون، وإذا ترملت أو طلقت وعادت لبيت أبيها تعود تأكل من الأقداس.{{p}}وما معنى هذا في الكنيسة ؟ الأقداس تشير لجسد المسيح ودمه ويتناول الأسرار المقدسة {{b}}الكاهن{{/b}} سواء الذي تمتع بسر الكهنوت، أو الكاهن بالمفهوم العام الذي ولد من المعمودية وهذا يسمى {{b}}مولود البيت{{/b}}. ويأكل منها {{b}}من إشترى بالفضة{{/b}} أي إقتناه الله بفدائه، ومن هم من أهل البيت فهم ليسوا زائرين بل يقضوا حياتهم كلها في بيت الرب. وتحذرنا الكنيسة من تقديم الذبيحة {{b}}لأجنبى{{/b}} أي إنسان تغرب عن الله، ورفض الشركة معه كإبن أو {{b}}لإنسان ثقب أذنيه{{/b}} أي عاش عبداً للخطية لا يريد أن يتحرر منها. أما {{b}}الإبنة التي{{/b}}{{b}}تزوجت بغريب{{/b}} فهي النفس التي قبلت الميلاد الجديد ثم عادت لتلتصق بعريس أجنبى أي بإله آخر (شهوة البطن / لذة الجسد / محبة المال / طلب الكرامة الزمنية). أما لو {{b}}مات{{/b}}{{b}}رجله{{/b}}ا أو طلقت أي تركت خطيتها تعود للوليمة المقدسة (حين عاد الابن الضال أكل العجل المسمن).{{p}}آية ١٤:- {{b}}إذا أكل سهواً{{/b}} = أما لو أكل عامداً مستهتراً بالوصية يقطع. وكان لهذا القانون إستثناء وافق الرب يسوع عليه حين أكل داود ورجاله الخبز المقدس (١صم٢١: ٦ + مت١٢: ٣ – ٧).{{p}}{{v16}}{{b}}أية (١٦):- "{{/b}}{{b}}١٦فَيُحَمِّلُونَهُمْ ذَنْبَ إِثْمٍ بِأَكْلِهِمْ أَقْدَاسَهُمْ. لأَنِّي أَنَا الرَّبُّ مُقَدِّسُهُمْ»."{{/b}}{{p}}{{b}}فيحملونهم ذنب إثمهم{{/b}} = أي يكونون بعدم إعتنائهم بمنع من لا يجوز له أن يأكل الأقداس فوق كونهم آثمين، سبباً لأن يحمل غيرهم الإثم بأكله من الأقداس بغير إستحقاق. وهذه الآية تثبت وجهة النظر التي تقول أن ما تناوله يهوذا ليلة تأسيس سر العشاء الربانى، لم يكن جسد المسيح ودمه، فالمسيح قطعا كان لا يريد أن يزيد أثام يهوذا، خصوصا أن التلاميذ في ذلك الوقت ما كانوا فاهمين تماما ما هو سر الإفخارستيا.{{p}}{{v17}}{{v18}}{{v19}}{{v20}}{{v21}}{{v22}}{{v23}}{{v24}}{{v25}}{{v26}}{{b}}الأيات (١٧-٢٥):-"{{/b}}{{b}}١٧وَكَلَّمَ الرَّبُّ مُوسَى قَائِلاً: ١٨«كَلِّمْ هَارُونَ وَبَنِيهِ وَجَمِيعِ بَنِي إِسْرَائِيلَ وَقُلْ لَهُمْ: كُلُّ إِنْسَانٍ مِنْ بَيْتِ إِسْرَائِيلَ وَمِنَ الْغُرَبَاءِ فِي إِسْرَائِيلَ، قَرَّبَ قُرْبَانَهُ مِنْ جَمِيعِ نُذُورِهِمْ وَجَمِيعِ نَوَافِلِهِمِ الَّتِي يُقَرِّبُونَهَا لِلرَّبِّ مُحْرَقَةً، ١٩فَلِلرِّضَا عَنْكُمْ يَكُونُ ذَكَرًا صَحِيحًا مِنَ الْبَقَرِ أَوِ الْغَنَمِ أَوِ الْمَعْزِ. ٢٠كُلُّ مَا كَانَ فِيهِ عَيْبٌ لاَ تُقَرِّبُوهُ لأَنَّهُ لاَ يَكُونُ لِلرِّضَا عَنْكُمْ. ٢١وَإِذَا قَرَّبَ إِنْسَانٌ ذَبِيحَةَ سَلاَمَةٍ لِلرَّبِّ وَفَاءً لِنَذْرٍ، أَوْ نَافِلَةً مِنَ الْبَقَرِ أَوِ الأَغْنَامِ، تَكُونُ صَحِيحَةً لِلرِّضَا. كُلُّ عَيْبٍ لاَ يَكُونُ فِيهَا. ٢٢الأَعْمَى وَالْمَكْسُورُ وَالْمَجْرُوحُ وَالْبَثِيرُ وَالأَجْرَبُ وَالأَكْلَفُ، هذِهِ لاَ تُقَرِّبُوهَا لِلرَّبِّ، وَلاَ تَجْعَلُوا مِنْهَا وَقُودًا عَلَى الْمَذْبَحِ لِلرَّبِّ. ٢٣وَأَمَّا الثَّوْرُ أَوِ الشَّاةُ الزَّوَائِدِيُّ أَوِ الْقُزُمُ فَنَافِلَةً تَعْمَلُهُ، وَلكِنْ لِنَذْرٍ لاَ يُرْضَى بِهِ. ٢٤وَمَرْضُوضَ الْخِصْيَةِ وَمَسْحُوقَهَا وَمَقْطُوعَهَا لاَ تُقَرِّبُوا لِلرَّبِّ. وَفِي أَرْضِكُمْ لاَ تَعْمَلُوهَا. ٢٥وَمِنْ يَدِ ابْنِ الْغَرِيبِ لاَ تُقَرِّبُوا خُبْزَ إِلهِكُمْ مِنْ جَمِيعِ هذِهِ، لأَنَّ فِيهَا فَسَادَهَا. فِيهَا عَيْبٌ لاَ يُرْضَى بِهَا عَنْكُمْ»."{{/b}}{{p}}هنا نجد شرط أن تكون الذبيحة بلا عيب فهي تشير للمسيح،  فالله لا يطلب كثرة الذبائح بل نوعيتها. وحين قال بيلاطس "إنى برئ من دم هذا البار" قدم المسيح كذبيحة بلا عيب. وكان هناك إستثناء واحد إن قدم أحدهم {{b}} نافلة {{/b}}كان ممكن أن يقدم {{b}} زوائدى{{/b}} أي من له عضو أطول من عضو أو إذا كان {{b}}قزما{{/b}} (قصير القامة جداً). والنافلة هي التي لم يلزمهم الله بها (ذبيحة خطية أو إثم....) أو ألزموا هم أنفسهم بها (نذر) أي تقدمة طوعية إختيارية. وهذا يظهر لنا الفارق، فالله حين قدم لنا دون إلزام وبحريته قدم ابنه القدوس. وهذا الزوائدى أو القزم لم يكن يقدم للهيكل كذبيحة بل يبيعونه ويعطوا ثمنه للهيكل. ولاحظ أن الوثنيين كان لهم عادة أن يُخصوا حيواناتهم والله منع هذا ومنع تقديم مثل هذه الحيوانات كذبيحة حتى لو كانوا قد إشتروها هكذا من أجنبى = {{b}} ومن يد ابن الغريب لا تقربوا خبز إلهكم{{/b}} = فالذبائح هي خبز الله لأن نار المذبح تأكله. حتى لا يظن اليهودى أن الله سيقبل منه ذبيحته لو كانت مخصية لأن الغريب فعل بها هذا.{{p}}{{b}}أية (٢٦):- "٢٦وَكَلَّمَ الرَّبُّ مُوسَى قَائِلاً: "{{/b}}{{p}}{{v27}}{{b}}أية (٢٧):- "{{/b}}{{b}}٢٧«مَتَى وُلِدَ بَقَرٌ أَوْ غَنَمٌ أَوْ مِعْزًى يَكُونُ سَبْعَةَ أَيَّامٍ تَحْتَ أُمِّهِ، ثُمَّ مِنَ الْيَوْمِ الثَّامِنِ فَصَاعِدًا يُرْضَى بِهِ قُرْبَانَ وَقُودٍ لِلرَّبِّ."{{/b}}{{p}}يشترط أن يقضى المولود ٧ أيام مع أمه قبل أن يقدم ذبيحة. ١)حتى يراعوا المشاعر فيعطوا فرصة للأم والابن أن يتمتعوا معاً ٢) أسبوع رقم كامل إشارة للمسيح الذي عاش وسطنا عمره كاملاً حتى أنهى عمله، فلم يقتله هيرودس طفلاً ولا قتله اليهود مرارا حينما دبروا قتله إذ كان يجتاز وسطهم وهم غير قادرين على المساس به. ثم صلب بعد أن أنهى عمله. ٣) اليهود فهموا هذا أنه حتى يمضى عليهم سبت، فالسبت مقدس، وبالتالي يتقدس قربانهم. {{p}}{{v28}}{{b}}أية (٢٨):- "{{/b}}{{b}}٢٨وَأَمَّا الْبَقَرَةُ أَوِ الشَّاةُ فَلاَ تَذْبَحُوهَا وَابْنَهَا فِي يَوْمٍ وَاحِدٍ."{{/b}}{{p}}{{b}}لا يقدم حيوان وأمه في يوم واحد{{/b}} = فالأم ستحزن حين يأخذوا ولدها، وهكذا الولد سيحزن ويصرخ دون إستجابة. وهكذا حدث مع المسيح "إلهي إلهي لماذا تركتني" والآب حزن لهذا. {{p}}{{v29}}{{b}}أية (٢٩):- "{{/b}}{{b}}٢٩وَمَتَى ذَبَحْتُمْ ذَبِيحَةَ شُكْرٍ لِلرَّبِّ، فَلِلرِّضَا عَنْكُمْ تَذْبَحُونَهَا."{{/b}}{{p}}أحسن شرح لها "نحن نحبه لأنه أحبنا أولاً".{{p}}{{v30}}{{v31}}{{v32}}{{v33}}{{b}}الأيات (٣٠-٣٣):-" ٣٠فِي ذلِكَ الْيَوْمِ تُؤْكَلُ. لاَ تُبْقُوا مِنْهَا إِلَى الْغَدِ. أَنَا الرَّبُّ. ٣١فَتَحْفَظُونَ وَصَايَايَ وَتَعْمَلُونَهَا. أَنَا الرَّبُّ. ٣٢وَلاَ تُدَنِّسُونَ اسْمِي الْقُدُّوسَ، فَأَتَقَدَّسُ فِي وَسَطِ بَنِي إِسْرَائِيلَ. أَنَا الرَّبُّ مُقَدِّسُكُمُ ٣٣الَّذِي أَخْرَجَكُمْ مِنْ أَرْضِ مِصْرَ لِيَكُونَ لَكُمْ إِلهًا. أَنَا الرَّبُّ»."{{/b}}{{p}}{{b}}
